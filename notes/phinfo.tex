    \begin{itemize}
    \item Initial state: $S^i_S=k_B \log2$
    \item Final state: $S^f_S=-k_B \log2$
    \end{itemize}
    \begin{align*}
    \Delta S_T &= \Delta S_S + \Delta S_E \ge 0 \\
    \Delta S_E &\geq - \Delta S_S \\ 
    \Delta S_E &\geq k_B\log 2\\
    \Delta Q_E &= T \Delta S_E \geq k_B T \log 2 \\
    \end{align*}
